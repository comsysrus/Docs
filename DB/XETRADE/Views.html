XETRADE.MY_PORT_VIEW

General Info

Caption MY_PORT_VIEW

Body 
AS
 SELECT mp.ID,
              ps.pseudo_name,
              mp.cap_part,
              mp.num_contr,
              mp.FIXED_LOTS,
              su.username,
              mp.port_id
         FROM XETRADE.my_port mp,
              XETRADE.my_port_head mph,
              XETRADE.system_users su,
              ORABET.PSEUDO_SYSTEM ps
        WHERE     mph.ID = mp.port_id
              AND su.ID = mph.owner_id
              AND ps.ID = mp.system_id
     ORDER BY pseudo_name
